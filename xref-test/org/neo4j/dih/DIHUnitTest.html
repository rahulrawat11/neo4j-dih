37      public ExpectedException thrown = ExpectedException.none();

39      /**
40       * Create an in memory graph database.
41       */
42      protected void initGraphDb() {
43          // Create Neo4j in memory database.
44          this.graphDb = new TestGraphDatabaseFactory()
45                  .newImpermanentDatabaseBuilder()
46                  .loadPropertiesFromFile("target/test-classes/neo4j.properties")
47                  .newGraphDatabase();
48      }
49  
50      /**
51       * Destroy the graph database.
52       */
53      protected void destroyGraphDb() {
54          this.graphDb.shutdown();
55      }
56  
57      /**
58       * Create an in memory graph database server.
59       */
60      protected void initServerWithExtension() throws IOException {
61          this.server = TestServerBuilders.newInProcessBuilder()
62                  .withExtension("/dih", DataImportHandlerExtension.class)
63                  .newServer();
64      }
65  
66      /**
67       * Destroy the in memory graph database server.
68       */
69      protected void destroyServerWithExtension() {
70          this.server.close();
71      }
72  
73      /**
74       * Create an in memrory H2 database.
75       */
76      protected void initH2() throws SQLException {
77          // Create H2 database
78          this.h2 = Server.createTcpServer().start();
79          Connection connection = DriverManager.getConnection("jdbc:h2:tcp://localhost/~/test;INIT=runscript from 'classpath:/datasource/jdbc/init.sql'", "SA", "");
80          connection.close();
81      }
82  
83      /**
84       * Destroy the H2 database.
85       */
86      protected void destroyH2() throws SQLException {
87          this.h2.stop();
88      }
89  
90      protected Result cypher(String script) {
91          Result rs = null;
92  
93          if (!StringUtils.isEmpty(script)) {
94              try (Transaction tx = graphDb.beginTx()) {
95                  rs = graphDb.execute(script);
96                  tx.success();
97              }
98          }
99          return rs;
100     }
