21  /**
22   * Server that parse the XML DIH config file.
23   *
24   * @author bsimard
25   * @version $Id: $Id
26   */
27  public class XmlParserService {
28  
29      /**
30       * The logger
31       */
32      private static final Logger log = LoggerFactory.getLogger(XmlParserService.class);
33  
34      /**
35       * The XML unmarshaller.
36       */
37      private Unmarshaller unmarshaller;
38  
39  
40      /**
41       * Constructor of the service.
42       * If something go wrong, it throw a RuntimeException, but this shouldn't happen.
43       */
44      public XmlParserService() {
45          try {
46              // Create the schema
47              File schemaFile = new File(ClassLoader.getSystemResource("schema/dataConfig.xsd").getFile());
48              SchemaFactory sf = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
49              Schema schema = sf.newSchema(schemaFile);
50  
51              // JAXB instance with <code>DataConfig.class</code> context
52              JAXBContext jc = JAXBContext.newInstance(DataConfig.class);
53  
54              // Create the unmarshaller
55              this.unmarshaller = jc.createUnmarshaller();
56              this.unmarshaller.setSchema(schema);
57  
58          } catch (JAXBException | SAXException e) {
59              throw new DIHRuntimeException(e);
60          }
61      }
62  
63      /**
64       * Method that parse an XML file.
65       *
66       * @param filename The DIH XML filename to parse
67       * @return a {@link generated.DataConfig} object.
68       * @throws org.neo4j.dih.exception.DIHException if any.
69       */
70      public DataConfig execute(String filename) throws DIHException {
71          File file = this.findConfigFileByName(filename);
72          try {
73              JAXBElement<DataConfig> root = unmarshaller.unmarshal(new StreamSource(file), DataConfig.class);
74              return root.getValue();
75          } catch (JAXBException e) {
76              throw new DIHException(e);
77          }
78      }
79  
80      /**
81       * Find the config file by its name.
82       * If no file is found, it return an <code>DIHException</code> exception.
83       *
84       * @param name a {@link java.lang.String} object.
85       * @return a {@link java.io.File} object.
86       * @throws org.neo4j.dih.exception.DIHException if any.
87       */
88      protected File findConfigFileByName(String name) throws DIHException {
89          if (ClassLoader.getSystemResource("conf/dih/" + name) == null) {
90              throw new DIHException("Config file %s doesn't exist.", name);
91          }
92          File file = new File(ClassLoader.getSystemResource("conf/dih/" + name).getFile());
93          if (!file.exists())
94              throw new DIHException("Config file %s doesn't exist.", name);
95          return file;
96      }
97  }
