1   package org.neo4j.dih.datasource;
2   
3   import generated.DataSourceType;
4   import generated.EntityType;
5   import org.neo4j.dih.exception.DIHException;
6   
7   import java.util.Map;
8   
9   /**
10   * Abstract class that represent a datasource.
11   *
12   * @author bsimard
13   * @version $Id: $Id
14   */
15  public abstract class AbstractDataSource {
16  
17      /**
18       * Datasource must have a constructor with a <code>DataSourceType</code>
19       *
20       * @param config Xml bean object that represent a datasource.
21       */
22      public AbstractDataSource(DataSourceType config){};
23  
24      /**
25       * This method is called when a job is starting. It permits to make some initialising stuff before starting a import job.
26       * For example, on <code>JDBCDataSource</code> we create a database connection.
27       *
28       * @throws org.neo4j.dih.exception.DIHException if any.
29       */
30      public abstract void start() throws DIHException;
31  
32      /**
33       * Datasource must have an <code>execute</code> method to perform the entity work.
34       *
35       * @param entity Xml bean Entity.
36       * @param state The current state of variables of the import.
37       * @throws org.neo4j.dih.exception.DIHException if any.
38       * @return a {@link org.neo4j.dih.datasource.AbstractResultList} object.
39       */
40      public abstract AbstractResultList execute(EntityType entity, Map<String, Object> state) throws DIHException;
41  
42      /**
43       * This method is called when a job is ending. It permits to close all open resources.
44       * For example, on <code>JDBCDataSource</code> we release the database connection.
45       *
46       * @throws org.neo4j.dih.exception.DIHException if any.
47       */
48      public abstract void finish() throws DIHException;
49  }
