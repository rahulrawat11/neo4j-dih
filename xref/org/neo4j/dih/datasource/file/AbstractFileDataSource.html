1   package org.neo4j.dih.datasource.file;
2   
3   import generated.DataSourceType;
4   import org.apache.commons.lang.StringUtils;
5   import org.neo4j.dih.datasource.AbstractDataSource;
6   import org.neo4j.dih.exception.DIHRuntimeException;
7   
8   import java.math.BigInteger;
9   
10  /**
11   * Just an abstraction for define file datasource type.
12   *
13   * @author bsimard
14   * @version $Id: $Id
15   */
16  public abstract class AbstractFileDataSource extends AbstractDataSource {
17  
18      /**
19       * Encoding of the file.
20       */
21      protected String encoding = "UTF-8";
22  
23      /**
24       * Url of the file.
25       */
26      protected String url;
27  
28      /**
29       * Timeout.
30       */
31      protected BigInteger timeout = BigInteger.valueOf(1000);
32  
33      /**
34       * Default constructor.
35       *
36       * @param config a {@link generated.DataSourceType} object.
37       */
38      public AbstractFileDataSource(DataSourceType config) {
39          super(config);
40          this.url = config.getUrl();
41          if (StringUtils.isEmpty(config.getUrl())) {
42              throw new DIHRuntimeException("File datasource url is mandatory.");
43          }
44          if (!StringUtils.isEmpty(config.getEncoding())) {
45              this.encoding = config.getEncoding();
46          }
47          if (config.getTimeout() != null) {
48              this.timeout = config.getTimeout();
49          }
50      }
51  
52  }
