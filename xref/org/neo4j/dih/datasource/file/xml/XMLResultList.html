23  /**
24   * ResultSet for {@link org.neo4j.dih.datasource.file.xml.XMLDataSource} object.
25   *
26   * @author bsimard
27   * @version $Id: $Id
28   */
29  public class XMLResultList extends AbstractResultList {
30  
31      /**
32       * The logger
33       */
34      private static final Logger log = LoggerFactory.getLogger(XMLResultList.class);
35  
36      /**
37       * Stream of the file.
38       */
39      private InputStream stream;
40  
41      /**
42       * List of entity return by xpath expression.
43       */
44      private NodeList nodeList;
45  
46      /**
47       * The current row cursor.
48       */
49      private Integer current = -1;
50  
51      /**
52       * Constructor.
53       *
54       * @param url       Url of the XML file
55       * @param timeout   Timeout
56       * @param encoding  Encoding of the XML file.
57       * @param query     For each XPATH query
58       * @throws org.neo4j.dih.exception.DIHException if any.
59       */
60      public XMLResultList(String url, BigInteger timeout, String encoding, String query) throws DIHException {
61          try {
62              // Get stream of file
63              URLConnection connection =new URL(url).openConnection();
64              connection.setConnectTimeout(timeout.intValue());
65              this.stream = connection.getInputStream();
66  
67              // Build the Xpath parser
68              DocumentBuilderFactory factory = DocumentBuilderFactory.newInstance();
69              DocumentBuilder builder = factory.newDocumentBuilder();
70              Document doc = builder.parse(stream);
71              XPathFactory xPathfactory = XPathFactory.newInstance();
72              XPath xpath = xPathfactory.newXPath();
73              XPathExpression expr = xpath.compile(query);
74              this.nodeList = (NodeList) expr.evaluate(doc, XPathConstants.NODESET);
75          } catch (Exception e) {
76              throw new DIHException(e);
77          }
78  
79      }
80  
81      /** {@inheritDoc} */
82      @Override
83      public boolean hasNext() {
84          return current < (nodeList.getLength() -1);
85      }
86  
87      /** {@inheritDoc} */
88      @Override
89      public XMLResult next() {
90          current++;
91          Node node = nodeList.item(current);
92          return new XMLResult(node);
93      }
94  
95      /** {@inheritDoc} */
96      @Override
97      public void close() throws IOException {
98          stream.close();
99      }
100 
101 }
